Create `core/runner.py`:

```python
import csv
import os

from config import CC_ADDRESSES
from core.generator import generate_certificate_bytes
from core.progress import ProgressTracker
from core.sender import AccountPool, send_one
from core.validator import clean_name


def run_tier(
    tier_name: str,
    tier: dict,
    pool: AccountPool | None,
    batch: int = 0,
    progress: ProgressTracker | None = None,
) -> dict:
    if progress is None:
        progress = ProgressTracker()

    if not os.path.exists(tier["template"]):
        raise FileNotFoundError(f"Template not found: {tier['template']}")

    with open(tier["csv"], newline="", encoding="utf-8") as fh:
        rows = list(csv.DictReader(fh))

    cursor = progress.get_cursor(tier_name)
    if cursor >= len(rows):
        return {"generated": 0, "sent": 0, "skipped": 0, "failed": 0}

    counts = {"generated": 0, "sent": 0, "skipped": 0, "failed": 0}
    smtp = pool.connect() if pool else None

    try:
        for idx in range(cursor, len(rows)):
            row = rows[idx]
            name = clean_name(row.get("name", ""))
            email = row.get("email", "").strip()

            if not name or not email:
                counts["skipped"] += 1
                progress.save_cursor(tier_name, idx + 1)
                continue

            if pool and progress.is_sent(email):
                progress.save_cursor(tier_name, idx + 1)
                continue

            pdf_bytes = generate_certificate_bytes(name, tier)
            counts["generated"] += 1

            if pool is None:
                progress.save_cursor(tier_name, idx + 1)
                continue

            try:
                filename = f"{name.lower().replace(' ', '_')}_{tier['cert_suffix']}.pdf"
                smtp = send_one(
                    smtp=smtp,
                    pool=pool,
                    from_addr=pool.current_user,
                    to_addr=email,
                    cc_addrs=CC_ADDRESSES,
                    subject=tier["email_subject"],
                    html_body=tier["email_html_body"].format(name=name),
                    plain_body=tier["email_plain_body"].format(name=name),
                    pdf_bytes=pdf_bytes,
                    filename=filename,
                )
                progress.mark_sent(tier_name, name, email)
                progress.save_cursor(tier_name, idx + 1)
                counts["sent"] += 1
                print(f"[{tier_name.upper()}] OK   {name} <{email}>  ({idx + 1}/{len(rows)})")

                if batch and counts["sent"] >= batch:
                    print(f"[{tier_name.upper()}] Batch of {batch} done. Re-run to continue.")
                    return counts
            except RuntimeError:
                raise
            except Exception as exc:
                progress.mark_failed(tier_name, name, email)
                counts["failed"] += 1
                print(f"[{tier_name.upper()}] FAIL {name} <{email}>: {exc}")
    finally:
        if smtp:
            try:
                smtp.quit()
            except Exception:
                pass

    return counts
```

```python
"""
Usage:
    certdrop run [tier] [--dry-run] [--batch=N]
    certdrop ui
    certdrop status
    certdrop retry
    certdrop init
"""
import os
import sys
import webbrowser

import click
from dotenv import load_dotenv

load_dotenv()

from config import TIERS
from core.progress import ProgressTracker
from core.runner import run_tier
from core.sender import AccountPool
from core.validator import validate_csv


def _load_pool() -> AccountPool:
    raw = os.getenv("SMTP_ACCOUNTS", "")
    if raw:
        accounts = [tuple(p.split(":", 1)) for p in raw.split(",") if ":" in p]
    else:
        user = os.getenv("GMAIL_USER")
        pwd = os.getenv("GMAIL_APP_PASSWORD")
        if not user or not pwd:
            raise click.ClickException("Set GMAIL_USER and GMAIL_APP_PASSWORD (or SMTP_ACCOUNTS) in .env")
        accounts = [(user, pwd)]
    limit = int(os.getenv("SMTP_PER_ACCOUNT_LIMIT", "450"))
    return AccountPool(accounts, per_account_limit=limit)


@click.group()
def cli():
    pass


@cli.command()
@click.argument("tier", required=False)
@click.option("--dry-run", is_flag=True, help="Generate certs only, skip sending")
@click.option("--batch", default=0, help="Stop after N sends (0 = no limit)")
def run(tier, dry_run, batch):
    """Generate and send certificates."""
    if tier and tier not in TIERS:
        raise click.ClickException(f"Unknown tier '{tier}'. Choose from: {', '.join(TIERS)}")

    pool = None if dry_run else _load_pool()
    progress = ProgressTracker()

    if not dry_run:
        click.echo("Connecting to SMTP…")

    for tier_name, tier_cfg in TIERS.items():
        if tier and tier_name != tier:
            continue
        counts = run_tier(tier_name, tier_cfg, pool=pool, batch=batch, progress=progress)
        if dry_run:
            click.echo(f"[{tier_name.upper()}] Generated {counts['generated']}, skipped {counts['skipped']} (dry-run)")
        else:
            click.echo(f"[{tier_name.upper()}] Sent {counts['sent']}, failed {counts['failed']}, skipped {counts['skipped']}")


@cli.command()
def status():
    """Show send progress for all tiers."""
    import json
    progress = ProgressTracker()
    click.echo(f"{'Tier':<20} {'Cursor':>8} {'Sent':>8} {'Failed':>8}")
    click.echo("-" * 48)
    for tier_name, tier_cfg in TIERS.items():
        import csv as _csv
        total = sum(1 for _ in open(tier_cfg["csv"], encoding="utf-8")) - 1
        cursor = progress.get_cursor(tier_name)
        sent = sum(1 for r in _csv.DictReader(open("sent_log.csv", encoding="utf-8")) if r.get("tier") == tier_name and r.get("status") == "sent") if os.path.exists("sent_log.csv") else 0
        failed = sum(1 for r in _csv.DictReader(open("retry.csv", encoding="utf-8")) if r.get("tier") == tier_name) if os.path.exists("retry.csv") else 0
        click.echo(f"{tier_name:<20} {cursor:>8}/{total:<6} {sent:>8} {failed:>8}")


@cli.command()
def retry():
    """Retry all failed sends."""
    progress = ProgressTracker()
    queue = progress.get_retry_queue()
    if not queue:
        click.echo("No failed sends to retry.")
        return

    pool = _load_pool()
    smtp = pool.connect()
    retried = 0

    for row in queue:
        tier_name = row["tier"]
        name = row["name"]
        email = row["email"]
        tier_cfg = TIERS.get(tier_name)
        if not tier_cfg:
            continue

        from core.generator import generate_certificate_bytes
        from core.sender import send_one
        from config import CC_ADDRESSES

        pdf_bytes = generate_certificate_bytes(name, tier_cfg)
        filename = f"{name.lower().replace(' ', '_')}_{tier_cfg['cert_suffix']}.pdf"
        try:
            smtp = send_one(
                smtp=smtp, pool=pool, from_addr=pool.current_user,
                to_addr=email, cc_addrs=CC_ADDRESSES,
                subject=tier_cfg["email_subject"],
                html_body=tier_cfg["email_html_body"].format(name=name),
                plain_body=tier_cfg["email_plain_body"].format(name=name),
                pdf_bytes=pdf_bytes, filename=filename,
            )
            progress.mark_sent(tier_name, name, email)
            progress.clear_retry_entry(email)
            click.echo(f"OK   {name} <{email}>")
            retried += 1
        except Exception as exc:
            click.echo(f"FAIL {name} <{email}>: {exc}")

    smtp.quit()
    click.echo(f"\nRetried {retried}/{len(queue)}")


@cli.command()
def ui():
    """Start the local web UI at http://localhost:5000."""
    import uvicorn
    from api.server import app
    webbrowser.open("http://localhost:5000")
    uvicorn.run(app, host="127.0.0.1", port=5000)


@cli.command()
def init():
    """Interactive setup wizard for a new event."""
    click.echo("CertDrop Setup Wizard\n")
    event_name = click.prompt("Event name")
    n_tiers = click.prompt("Number of tiers", type=int, default=1)
    click.echo(f"\nConfigured {n_tiers} tier(s) for '{event_name}'.")
    click.echo("Edit config.py to set templates, CSVs, and email copy.")
    click.echo("Then run: certdrop ui")


if __name__ == "__main__":
    cli()
```
